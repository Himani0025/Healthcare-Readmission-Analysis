set sql_mode = '';

select * from healthcare limit 10;

select count(*) from healthcare;


-- check null values
select count(*)
from healthcare
where readmitted_flag is null;


-- check duplicates (if encounter_id exists)

select encounter_id, count(*)
from healthcare
group by encounter_id
having count(*) > 1;

# Basic Exploratory Queries

-- Readmission Distribution

select readmitted_flag, count(*) as total
from healthcare
group by readmitted_flag;


-- Average Time in Hospital

select avg(time_in_hospital)
from healthcare;


-- Readmission by Age

select age, readmitted_flag, count(*) as total
from healthcare
group by age, readmitted_flag
order by age;


-- readmission by number of medications

select readmitted_flag, avg(num_medications)
from healthcare
group by readmitted_flag;


# Readmission rate (KPI)

select
    (select count(*) from healthcare where readmitted_flag = "Yes") * 100 
    /
    (select count(*) from healthcare) as readmission_rate;
    
                  -- OR --
                  
select
	sum(case when readmitted_flag = "Yes"
    then 1 else 0 end) * 100.0 / count(*) 
    as readmission_rate
    from healthcare;
    

# Top factors Affecting readmission

-- Time in Hospital Impact

select readmitted_flag,
avg(time_in_hospital) as avg_stay
from healthcare
group by readmitted_flag;


-- Medication Impact

select readmitted_flag,
avg(num_medications) as avg_meds
from healthcare
group by readmitted_flag;

-- difference rate

SELECT 
(
    (SELECT AVG(num_medications) 
     FROM healthcare 
     WHERE readmitted_flag = 'Yes')
    -
    (SELECT AVG(num_medications) 
     FROM healthcare 
     WHERE readmitted_flag = 'No')
) 
* 100.0
/
(SELECT AVG(num_medications) 
 FROM healthcare 
 WHERE readmitted_flag = 'No')
AS percentage_higher;


-- Emergency Visits Impact

select readmitted_flag,
avg(number_emergency) as avg_emergency_visits
from healthcare
group by readmitted_flag;


-- High Risk Age

select age, 
sum(case when readmitted_flag = "yes"
then 1 else 0 end) * 100 / count(*) as readmission_rate
from healthcare
group by age
order by readmission_rate desc;
